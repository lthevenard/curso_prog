In[40]:
# gabarito
minha_paleta = {"yellow": "tab:olive", "green": "tab:green"}

fig, ax = plt.subplots()

sns.stripplot(
    x="tip",
    y="pickup_borough",
    hue="color",
    dodge=True,
    data=taxis,
    palette=minha_paleta,
    ax=ax  
)

ax.set_title("Distribuio de gorjetas por bairro de origem e cor do txi")
ax.set_xlabel("Valor da gorjeta (US$)")
ax.set_ylabel("Bairro de origem")
